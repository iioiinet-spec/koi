1. 알고리즘 핵심 전략

이 문제는 트리 구조에서 발생하는 대규모 쿼리를 처리해야 합니다. 집의 개수 N과 명령의 개수 Q가 모두 100,000 이하이므로, 단순한 DFS 탐색(O(N))을 매 명령마다 반복하면 총 100억 번의 연산이 필요하여 시간 초과가 발생합니다.

최적화의 열쇠: 오일러 투어 테크닉 (ETT) 트리를 DFS 순서로 탐색하여 진입 시간(in)과 탈출 시간(out)을 기록하면, 임의의 노드 u를 루트로 하는 서브트리는 배열 상에서 [in[u], out[u]]라는 연속된 구간이 됩니다.

2. 단계별 구현 로직

  • 트리 일렬화: DFS를 통해 각 노드의 서브트리 범위를 배열 인덱스로 매핑합니다.[cite: 5]
  • 데이터 구조: 수시로 변하는 구간 합을 관리하기 위해 세그먼트 트리를 사용합니다.[cite: 5]
  • 업데이트: 노드 u의 값 변경 시, in[u] 위치의 값을 세그먼트 트리에서 갱신합니다.[cite: 5]
  • 쿼리: 서브트리 합 요청 시, query(in[u], out[u])를 수행하여 O(log N)에 결과를 얻습니다.[cite: 5]

3. 정답 소스 코드 (C Language)

이 코드는 오일러 투어 테크닉과 세그먼트 트리를 결합하여 O((N + Q) log N)의 복잡도로 동작합니다.[cite: 5]


#include <stdio.h>

#define MAXN 100005

int N, Q, timer;
int head[MAXN], to[MAXN * 2], nxt[MAXN * 2], edgeCount;
int in[MAXN], out[MAXN];
long long value[MAXN], arr[MAXN], seg[MAXN * 4];

// [1] 인접 리스트로 트리 구성[cite: 5]
void addEdge(int u, int v) {
    to[edgeCount] = v;
    nxt[edgeCount] = head[u];
    head[u] = edgeCount++;
}

// [2] DFS를 이용한 오일러 투어 테크닉[cite: 5]
void dfs(int cur, int prev) {
    in[cur] = ++timer; // 구간 시작 위치 기록
    for (int e = head[cur]; e != -1; e = nxt[e]) {
        if (to[e] != prev) dfs(to[e], cur);
    }
    out[cur] = timer; // 구간 끝 위치 기록
}

// [3] 세그먼트 트리 구축 (초기화)[cite: 5]
void build(int node, int start, int end) {
    if (start == end) {
        seg[node] = arr[start];
        return;
    }
    int mid = (start + end) / 2;
    build(node * 2, start, mid);
    build(node * 2 + 1, mid + 1, end);
    seg[node] = seg[node * 2] + seg[node * 2 + 1];
}

// [4] 세그먼트 트리 업데이트 (값 변경)[cite: 5]
void update(int node, int start, int end, int idx, long long newVal) {
    if (start == end) {
        seg[node] = newVal;
        return;
    }
    int mid = (start + end) / 2;
    if (idx <= mid) update(node * 2, start, mid, idx, newVal);
    else update(node * 2 + 1, mid + 1, end, idx, newVal);
    seg[node] = seg[node * 2] + seg[node * 2 + 1];
}

// [5] 세그먼트 트리 쿼리 (구간 합)[cite: 5]
long long query(int node, int start, int end, int left, int right) {
    if (right < start || end < left) return 0;
    if (left <= start && end <= right) return seg[node];
    int mid = (start + end) / 2;
    return query(node * 2, start, mid, left, right) + 
           query(node * 2 + 1, mid + 1, end, left, right);
}

int main() {
    scanf("%d %d", &N, &Q);
    for (int i = 1; i <= N; i++) head[i] = -1;
    for (int i = 1; i <= N; i++) scanf("%lld", &value[i]);
    for (int i = 0; i < N - 1; i++) {
        int u, v;
        scanf("%d %d", &u, &v);
        addEdge(u, v); addEdge(v, u);
    }

    dfs(1, 0); // 트리 일렬화 수행[cite: 5]
    for (int i = 1; i <= N; i++) arr[in[i]] = value[i];
    build(1, 1, N); // 트리 생성[cite: 5]

    while (Q--) {
        int type, u;
        scanf("%d %d", &type, &u);
        if (type == 1) {
            long long x;
            scanf("%lld", &x);
            update(1, 1, N, in[u], x); // 값 수정[cite: 5]
        } else {
            // 서브트리 범위 [in[u], out[u]] 합 출력[cite: 5]
            printf("%lld\n", query(1, 1, N, in[u], out[u]));
        }
    }
    return 0;
}
